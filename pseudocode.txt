# see if user earns interest/dividends on their savings

while True
    MASUKKAN hasDividends
    PANGGIL pretendLoading(1)
    JIKA hasDividends != "y" dan hasDividends != "n"
        PAPAR "Please enter 'y' or 'n'."
        SAMBUNG
    LAIN JIKA hasDividends == "y"
        SET hasDividends = True
        KELUAR GELUNG
    JIKA TIDAK
        SET hasDividends = False
        KELUAR GELUNG
    TAMAT JIKA

# get user's interest rate/dividend (annual)
JIKA hasDividends
    while True
        MASUKKAN appreciationRate
        CETAK '\033[F' + '\r' + appreciationRate + "% p.a.\n"
        PANGGIL pretendLoading(1)
        CUBA
            # validate and convert to decimal
            SET appreciationRateDecimal = appreciationRate / 100 dalam Decimal
            JIKA appreciationRateDecimal < 0 atau appreciationRateDecimal == 0
                PAPARKAN "Enter a number greater than 0."
                SAMBUNG
            KELUAR GELUNG
            TAMAT JIKA
        JIKA terdapat ralat InvalidOperation
            PAPAR "Enter a valid number! Not text!"
            SAMBUNG
        TAMAT CUBA

# see how many times it compounds annually
JIKA hasDividends
    while True

# calculate how long it will take to reach the goal

#### doesnt have dividends

## formula is t = goalamt / income (result will be in months)

JIKA TIDAK hasDividends
    SET monthsToGoal = goalamtDecimal / incomeDecimal dan bundarkan
    SET yearsToGoal = monthsToGoal BAHAGI TANPA BAKI 12
    SET monthsToGoalWithoutYear = monthsToGoal MOD 12 dan bundarkan
    PANGGIL pretendLoading(1)
    PANGGIL printTimeToGoal()
TAMAT JIKA

### has dividends, with compounding

## formula is t = (ln(1 + (goalamt * r) / income)) / (ln(1 + r))

JIKA hasDividends
    SET r = appreciationRateDecimal / compoundingFrequencyInt
    SET numerator = ln((goalamtDecimal * r) / incomeDecimal + 1)
    SET denominator = ln(1 + r)
    SET t = numerator / denominator
    SET t = t * 12 / compoundingFrequencyInt

    SET monthsToGoal = pemboleh ubah t yang dibundar
    SET yearsToGoal = monthsToGoal BAHAGI TANPA BAKI 12
    SET monthsToGoalWithoutYear = (monthsToGoal % Decimal(12)) yang dibundar
    PANGGIL pretendLoading(1)
    PANGGIL printTimeToGoal()
